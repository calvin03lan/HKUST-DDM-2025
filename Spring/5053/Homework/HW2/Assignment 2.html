cand = []
for d, D in product([0, 1], repeat=2):
    for p, q, P, Q in product(range(0, 4), repeat=4):
        if p + q + P + Q == 0 or p + q + P + Q > 6:

            res = ARIMA(
                y, order=(p, d, q), seasonal_order=(P, D, Q, 4), trend="n"
