The potential energy of the particle is given by
$$ U(x) = \frac{1}{2} k x^2. $$

In thermal equilibrium, the probability density function for the particle’s position is proportional to the Boltzmann factor:
$$ f(x) \propto \exp\left(-\frac{U(x)}{k_B T}\right) = \exp\left(-\frac{k x^2}{2 k_B T}\right). $$

The problem specifies that $k_B T/k= 1 $. Substituting this condition yields
$$ f(x) \propto \exp\left(-\frac{1}{2} x^2\right). $$

This is the unnormalized form of the standard normal distribution. Normalizing gives
$$ f(x) = \frac{1}{\sqrt{2\pi}} \exp\left(-\frac{x^2}{2}\right), $$ which is the probability density function of ( \mathcal{N}(0, 1) ).

Consequently, the theoretical moments are:
$$ \langle x \rangle = 0 $$ $$ \langle x^2 \rangle = 1 $$ $$ \langle x^3 \rangle = 0 $$ $$ \langle x^4 \rangle = 3 $$

import numpy as np
import matplotlib.pyplot as plt

# Parameters
delta = 1.0          # proposal step size
N_burn = 1000        # burn-in steps
N_samples = 100000   # number of samples after burn-in

# Initialize
x = 0.0
samples = []

np.random.seed(42)   # for reproducibility

# Metropolis algorithm
for i in range(N_burn + N_samples):
    # Propose new state: symmetric uniform proposal in [x - delta, x + delta]
    x_prop = x + delta * (2 * np.random.rand() - 1)
    
    # Compute acceptance probability
    # f(x) ∝ exp(-x^2 / 2), so ratio = exp(-(x_prop^2 - x^2)/2)
    log_ratio = -0.5 * (x_prop**2 - x**2)
    accept_prob = min(1.0, np.exp(log_ratio))
    
    # Accept or reject
    if np.random.rand() < accept_prob:
        x = x_prop  # accept
    
    # Store sample after burn-in
    if i >= N_burn:
        samples.append(x)

samples = np.array(samples)

# Compute expectation values
mean_x   = np.mean(samples)
mean_x2  = np.mean(samples**2)
mean_x3  = np.mean(samples**3)
mean_x4  = np.mean(samples**4)

print(f"<x>   = {mean_x:.5f}   (theory: 0)")
print(f"<x^2> = {mean_x2:.5f}   (theory: 1)")
print(f"<x^3> = {mean_x3:.5f}   (theory: 0)")
print(f"<x^4> = {mean_x4:.5f}   (theory: 3)")

# Plot histogram vs theoretical PDF
x_vals = np.linspace(-4, 4, 400)
pdf_theory = (1 / np.sqrt(2 * np.pi)) * np.exp(-0.5 * x_vals**2)

plt.figure(figsize=(8, 5))
plt.hist(samples, bins=100, density=True, alpha=0.6, label='Metropolis samples')
plt.plot(x_vals, pdf_theory, 'r-', lw=2, label='Theoretical PDF (N(0,1))')
plt.xlabel('x')
plt.ylabel('Probability density')
plt.title('Histogram of Metropolis Samples vs Theoretical PDF')
plt.legend()
plt.grid(True, linestyle='--', alpha=0.5)
plt.tight_layout()
plt.show()
