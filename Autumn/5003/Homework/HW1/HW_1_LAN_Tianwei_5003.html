# (b) Fitting normal distribution

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm

def Fitting_Norm(mu_x:int, sigma_sq_x:int, mu_y:int, sigma_sq_y:int, n_samples: int = 1000):

    # Set distribution of X and Y
    X = np.random.normal(mu_x, np.sqrt(sigma_sq_x), n_samples)
    Y = np.random.normal(mu_y, np.sqrt(sigma_sq_y), n_samples)
    Z = X + Y

    # Fitting
    mu_z_hat, sigma_z_hat = norm.fit(Z)

    # Plot
    plt.hist(Z, bins=80, density=True, alpha=0.6, color='skyblue', label='Sample histogram')
    x = np.linspace(Z.min(), Z.max(), 200)
    plt.plot(x, norm.pdf(x, mu_z_hat, sigma_z_hat), 'r-', label=f'Fit: μ={mu_z_hat:.2f}, σ²={sigma_z_hat:.2f}**2')
    plt.title('Fitting Z = X + Y with Normal Distribution')
    plt.xlabel('Z')
    plt.ylabel('Density')
    plt.legend()
    plt.grid(True)
    plt.show()

    mu_z_theory = mu_x + mu_y
    sigma_z_theory = np.sqrt(sigma_sq_x + sigma_sq_y)

    print(f"Theoretical: μz = {mu_z_theory}, σ²z = {sigma_z_theory**2:.0f}")
    print(f"Fitted:      μ̂z = {mu_z_hat:.2f}, σ̂²z = {sigma_z_hat**2:.2f}")

if __name__ == "__main__":
    Fitting_Norm(0, 1, 1, 2, 1000)
    Fitting_Norm(0, 1, 1, 2, 10000)
    Fitting_Norm(0, 1, 1, 2, 100000)
